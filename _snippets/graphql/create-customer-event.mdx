```graphql Mutation
mutation createCustomerEvent($input: CreateCustomerEventInput!) {
  createCustomerEvent(input: $input) {
    customerEvent {
      id
      title
      createdAt {
        iso8601
        unixTimestamp
      }
    }
    error {
      message
      type
      code
      fields {
        field
        message
        type
      }
    }
  }
}
```
